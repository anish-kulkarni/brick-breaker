//////////// Classes /////////////////
Class Node_Gene{
	int type;
	int innovation_number;
}
Class Connection_Gene{
	int input;
	int output;
	double weight;
	boolean enabled;
	int innovation_number;
}

Class Connection_pool
{
	int start;
	int end;
	int innovation_number;
}

Class Node_pool
{
	int broken_connection_number;
	int innovation_number;
}

Class Genome{
	ArrayList<Connection_Gene> connection_genes
	ArrayList<Node_Gene> node_genes;
}

Class Organism{ //individual
	Genome genome;
	double fitness;
	int allowed_offsprings = 2;
}

Class Species {

	Array<Integers> individual // Array of indices of the offsprings that belong to the species
	double avg_corrected_fitness;
	int permitted_offsprings;
	int stagnant_period;
}

Class Generation{
	Array<Organisms> offspring;
	Array<Species> species;
	Array<Connection_pool> connection_gene_pool;
	Array<Node_pool> node_gene_pool;

}}
